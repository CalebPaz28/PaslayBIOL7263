  select(common_name, month, year, location, count) %>% 
  group_by(location, common_name, month, year, count) %>% 
  summarise(location)
plot_1 <- ggplot(data = count_bird) +
  aes(month, count) +
  geom_point(aes(color = year), size = 1)+
  facet_wrap(~location) +
